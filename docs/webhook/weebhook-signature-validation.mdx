# Como validar o Webhook Signature?

Toda request do Webhook possui o header 'x-webhook-signature' que é a signature gerada com a secret key da Woovi e o payload do Webhook.
Ao receber o header você pode validar se o signature é válido e dar continuidade no fluxo do Webhook.

Exemplo: `'x-webhook-signature': 'K6GVzzXzNjt7vpfzOlt/fgsZl0o88k4tR/ABuWd6u03YMYTchuUsXtLZCzzJOVBy6nkoj3SNLLFcMttno2/hgMHuVe+5rWNBDA4c6IaiXhAPnsIFA357xzLKWaEx9lRQJ66r8QA74MZczfLoaZsoyMROA/YC6PoBZERsRXxt/D0='`

import crypto from 'crypto';

type VerifyPayloadType = {
  payload: string;
  signature: string;

const algorithm = 'sha256';
const signatureFormat = 'base64';

export const verifyPayload = ({ payload, signature }: VerifyPayloadType) => {
  const publicKeyBase64 =
    'LS0tLS1CRUdJTiBQVUJMSUMgS0VZLS0tLS0KTUlHZk1BMEdDU3FHU0liM0RRRUJBUVVBQTRHTkFEQ0JpUUtCZ1FEVDdWYStxb3pvT2NYUStjSHJWNk85RVE0TgpnZVhvY1ZwRFBBWkpTZVJsbEVlQVVha051MURqY3FweDFmb1l5aEZxRTM3TkNWYzRtK0hvTC9nN1k3VDMyZVJ4CjhpandxMjdoY0ZjL0RFc01ISWdVU0U4cGdPbi96a3ZadXdNb256MkVjdy85NzZzTlUzNnpKOXhMUE53dURnSysKb2dUb0RQTmNkaWtRdi9STHFRSURBUUFCCi0tLS0tRU5EIFBVQkxJQyBLRVktLS0tLQo=';

  const verify = crypto.createVerify(algorithm);

  verify.end();

  return isValid;
};
